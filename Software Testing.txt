package com.test;

import org.openqa.selenium.By;
import org.openqa.selenium.Dimension;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.WindowType;
import org.openqa.selenium.chrome.ChromeDriver;
import org.testng.annotations.AfterMethod;
import org.testng.annotations.BeforeMethod;
import org.testng.annotations.Test;

public class NewTest {
	WebDriver wd;
  @Test 
  public void f() throws InterruptedException {
	  
	  String url = wd.getCurrentUrl();
	  System.out.println(url);
	  String title = wd.getTitle();
	  System.out.println(title);
	  Thread.sleep(3000);
	  
	  WebElement name = wd.findElement(By.id("name"));
	  name.sendKeys("Nadeer");
	  System.out.println(name.isEnabled());
	  System.out.println(name.isDisplayed());
	  System.out.println(name.getAttribute("value"));
	  
	  WebElement email = wd.findElement(By.id("email"));
	  email.sendKeys("nadeer@gmail.com");
	  System.out.println(email.isEnabled());
	  System.out.println(email.isDisplayed());
	  System.out.println(email.getAttribute("value"));
	  
	  Dimension d = new Dimension(520,750);
	  wd.manage().window().getSize();
	  System.out.println(d.getHeight());
	  System.out.println(d.getWidth());
	  
	}

  
  @BeforeMethod
  public void openLink() throws InterruptedException
  {
	  wd = new ChromeDriver();
	  String cw = wd.getWindowHandle();
	  wd.get("https://www.tutorialspoint.com/selenium/practice/selenium_automation_practice.php");
	  wd.switchTo().newWindow(WindowType.TAB);
	  
	  wd.get("https://www.google.com/");
	  wd.close();
	  Thread.sleep(3000);
	  wd.switchTo().window(cw);
	  
  }
  
  @AfterMethod
  public void closeLink()
  {
	  wd.close();
  }
}
